17'''
18@brief def verarbeite_transport Verarbeitet und überprüft die Transportdaten auf Fehler.
19@details
20
21 Allgeme Beschreibung:
22
23 Diese Funktion führt verschiedene Prüfungen auf den Ablauf
24 eines Transports durch (z. B. Übergabezeiten, Transportdauer,
25 doppelte OUTs, fehlende OUTs, GVZ/KT-Reihenfolge).
26 Alle gefundenen Auffälligkeiten werden als Textmeldungen
27 zurückgegeben.
28
29 Eine Liste mit Fehler-Meldungen (Strings):
30 - Wenn Fehler gefunden wurden: entsprechende Fehlermeldungen.
31 - Wenn keine Fehler gefunden wurden: ["korrekt"].
32
33 Schritt 1: Kein Transport vorhanden
34
35 Wenn in transport_daten keine Einträge gespeichert sind,
36 gibt es keinen Transport. In diesem Fall wird sofort die
37 Meldung "Es gibt gar keinen Eintrag" zurückgegeben und
38 die Funktion beendet.
39
40 Schritt 2: Daten sammeln und sortieren
41
42 Alle Transport-Einträge werden aus transport_daten herausgesucht,
43 in eine Liste "transport_daten_liste" gepackt und zeitlich sortiert.
44 Dazu gibt es zwei Hilfsfunktionen:
45 - get_status(): liefert 'in' oder 'out' zurück.
46 - get_art(): bestimmt die Art der Station (GVZ oder KT).
47
48 Schritt 3: Übergabe > 10 min
49
50 Nach jedem OUT-Ereignis wird die Zeit bis zum nächsten IN gemessen.
51 Liegt dieser Abstand über 10 Minuten, wird die Meldung
52 "Übergabe > 10 min" erstellt. Danach wird der OUT-Zeitpunkt zurückgesetzt.
53
54 Schritt 4: Transportdauer > 48h
55
56 Es wird die Zeit zwischen dem ersten IN und dem letzten OUT berechnet.
57 Überschreitet die Transportdauer 48 Stunden, wird die Meldung
58 "Transportdauer > 48h" hinzugefügt.
59
60 Schritt 5: Doppelter Auscheck-Zeitpunkt
61
62 Hier wird geprüft, ob an derselben Station zweimal direkt
63 hintereinander ein OUT gespeichert wurde. Wenn ja, wird der
64 Zeitunterschied berechnet und in der Meldung ausgegeben.
65
66 Schritt 6: Fehlende OUTs
67
68 Falls ein Transport an einer Station mit "in" endet, aber
69 kein "out" folgt, wird dies überprüft:
70 - Am Ende des Transports -> Meldung "Auscheck-Zeitpunkt fehlt am Ende".
71 - Mitten im Transport -> Meldung "Auscheck-Zeitpunkt fehlt in der Mitte".
72
73 Schritt 7: Aus und wieder Einchecken im gleichen Kühllager
74
75 Wurde ein Zyklus (IN -> OUT) an einer GVZ-Station abgeschlossen
76 und danach erfolgt erneut ein IN an derselben Station,
77 wird eine Meldung hinzugefügt.
78
79 Schritt 8: GVZ vor KT prüfen
80
81 Wenn ein KT noch aktiv ist (KT-IN ohne KT-OUT) und in dieser Zeit
82 ein GVZ-IN passiert, stimmt die Reihenfolge nicht. In diesem Fall
83 wird eine Meldung erstellt.
84
85 Schritt 9: Keine Fehler gefunden
86
87 Wenn nach allen Prüfungen keine Meldungen in der Liste stehen,
88 wird stattdessen "korrekt" zurückgegeben.
89
90'''

92def verarbeite_transport(transport_daten, temperatur_daten, company_daten, transportstation_daten):
93 meldungen = []
94
95 # #########################################
96 # 1. Kein Transport vorhanden
97 # #########################################
98
99 if len(transport_daten) == 0:
100 meldungen.append("Es gibt gar keinen Eintrag")
101 return meldungen
102
103 # #########################################
104 # 2. Daten sammeln und sortieren
105 # #########################################
106
107 transport_daten_liste = []
108
109 for key in transport_daten:
110 transport_daten_liste.append(transport_daten[key])
111 transport_daten_liste.sort(key=lambda x: x[5])
112
113 def get_status(eintrag):
114 return str(eintrag[4]).replace("'", "").lower()
115
116 def get_art(station_id):
117 for key in transportstation_daten:
118 if transportstation_daten[key][0] == station_id:
119 return str(transportstation_daten[key][2]).replace("'", "").upper()
120 return ""
121
122 # #########################################
123 # 3. Übergabe > 10 min
124 # #########################################
125
126 letzte_out = None
127
128 for bewegung in transport_daten_liste:
129 status = get_status(bewegung)
130 zeit = bewegung[5]
131 if status == "out":
132 letzte_out = zeit
133 if status == "in" and letzte_out != None:
134 diff = (zeit - letzte_out).total_seconds() / 60
135 if diff > 10:
136 meldungen.append("Übergabe > 10 min")
137 letzte_out = None
138
139 # ##########################################
140 # 4. Transportdauer > 48h
141 # ##########################################
142
143 erstes_in = None
144 letztes_out = None
145
146 for bewegung in transport_daten_liste:
147 status = get_status(bewegung)
148 zeit = bewegung[5]
149 if status == "in" and erstes_in == None:
150 erstes_in = zeit
151 if status == "out":
152 letztes_out = zeit
153 if erstes_in != None and letztes_out != None:
154 stunden = (letztes_out - erstes_in).total_seconds() / 3600
155 if stunden > 48:
156 meldungen.append("Transportdauer > 48h")
157
158 # ##########################################
159 # 5. Doppelter Auscheck-Zeitpunkt
160 # ##########################################
161
162 letzte_aktion = {}
163
164 for bewegung in transport_daten_liste:
165 station = bewegung[3]
166 status = get_status(bewegung)
167 zeit = bewegung[5]
168 if station in letzte_aktion:
169 if letzte_aktion[station][0] == "out" and status == "out":
170 diff = int((zeit - letzte_aktion[station][1]).total_seconds() / 60)
171 meldungen.append("Doppelter Auscheck-Zeitpunkt (Abstand " + str(diff) + " min)")
172 letzte_aktion[station] = [status, zeit]
173
174 # ##########################################
175 # 6. Fehlende OUTs - Konsitenzprüfung
176 # ##########################################
177
178 letzter_status = get_status(transport_daten_liste[-1])
179 fehlt_in_mitte = False
180
181 for station in letzte_aktion:
182 if letzte_aktion[station][0] == "in":
183 if letzter_status == "in":
184 meldungen.append("Auscheck-Zeitpunkt fehlt am Ende (kein Fehler., da nicht abgeschlossen.)")
185 else:
186 fehlt_in_mitte = True
187 if fehlt_in_mitte:
188 meldungen.append("Auscheck-Zeitpunkt fehlt in der Mitte")
189
190 # ##########################################
191 # 7. Aus und wieder Einchecken im gleichen Kühllager - Konsitenzprüfung
192 # ##########################################
193
194 zyklus_gesehen = {} # station_id -> True/False
195 letzter_status_station = {} # station_id -> 'in'/'out'
196
197 for bewegung in transport_daten_liste:
198 station = bewegung[3]
199 status = get_status(bewegung)
200 art = get_art(station)
201
202 if station not in zyklus_gesehen:
203 zyklus_gesehen[station] = False
204
205 if station in letzter_status_station:
206 if letzter_status_station[station] == "in" and status == "out":
207 zyklus_gesehen[station] = True
208
209 if status == "in" and zyklus_gesehen[station] and art == "GVZ":
210 if "Aus und wieder Einchecken im gleichen Kühllager" not in meldungen:
211 meldungen.append("Aus und wieder Einchecken im gleichen Kühllager")
212
213 letzter_status_station[station] = status
214
215 # ##########################################
216 # 8. GVZ vor KT prüfen - Konsitenzprüfung
217 # ##########################################
218
219 kt_aktiv = False
220
221 for bewegung in transport_daten_liste:
222 art = get_art(bewegung[3])
223 status = get_status(bewegung)
224
225 if art == "KT":
226 if status == "in":
227 kt_aktiv = True
228 elif status == "out":
229 kt_aktiv = False
230
231 elif art == "GVZ" and status == "in":
232 if kt_aktiv and "Einchecken Kühllager liegt zeitlich vor Auschecken LKW" not in meldungen:
233 meldungen.append("Einchecken Kühllager liegt zeitlich vor Auschecken LKW")
234
235 # ##########################################
236 # 9. Wenn keine Fehler -> korrekt
237 # ##########################################'
238
239 if len(meldungen) == 0:
240 meldungen.append("korrekt")
241
242 return meldungen
