129
130 def on_pruefen(self):
131
132 '''
133 @brief Führt die Prüfung eines einzelnen Transports basierend auf der eingegebenen ID durch.
134
135 @details Liest die Transport-ID aus dem Eingabefeld, ruft alle notwendigen Daten
136 (Transport, Temperatur, Unternehmen, Station) aus der Datenbank ab und
137 übergibt diese zur Verarbeitung. Die generierten Meldungen werden im
138 Ausgabefeld angezeigt. Stellt Status-Updates bereit.
139 '''
140
141 transportid = self.entry.get().strip()
142 if not transportid:
143 messagebox.showerror("Fehler", "Bitte eine Transport-ID eingeben!")
144 return
145
146 try:
147 # Status aktualisieren: Prüfung läuft
148 self.status.set("Prüfe Transport-ID...")
149 self.root.update_idletasks()
150
151
152 # Datenbankzugriffe
153 transport_daten, _ = get_transport_daten(transportid, verbindungs_i)
154 temperatur_daten, _ = get_temperatur_daten(transport_daten, verbindungs_i)
155 company_daten, _ = get_company_daten(transport_daten, verbindungs_i)
156 transportstation_daten, _ = get_transportstation_daten(transport_daten, verbindungs_i)
157
158 # Verarbeitung
159 meldungen = verarbeite_transport(
160 transport_daten, temperatur_daten, company_daten, transportstation_daten
161 )
162
163 # Ausgabe im Textfeld
164 self.output.delete("1.0", tk.END)
165 self.output.insert(tk.END, f"Transport-ID: {transportid}\n")
166 self.output.insert(tk.END, "--------------------------------------------------\n")
167 for m in meldungen:
168 self.output.insert(tk.END, f" -> {m}\n")
169
170 #Status aktualisieren
171 self.status.set("Prüfung abgeschlossen")
172
173 except Exception as e:
174 messagebox.showerror("Fehler", f"Ein Fehler ist aufgetreten:\n{e}")
175
176 # ##################################################################
177 # Funktion: Alle Transporteid meldungen ausgeben
178 # ##################################################################
179
180 def on_alle_pruefen(self):
181
182 '''
183 @brief Führt die Prüfung aller in der Datenbank vorhandenen Transporte durch.
184
185 @details Ruft zunächst alle eindeutigen Transport-IDs ab. Iteriert dann
186 über jede ID, holt die zugehörigen Daten, verarbeitet sie und sammelt
187 alle generierten Meldungen. Die Ergebnisse werden nach Meldungstext sortiert
188 und im Ausgabefeld angezeigt. Stellt Status-Updates bereit.
189 '''
190
191 try:
192 # Status aktualisieren: Prüfung läuft
193 self.status.set("Bitte warten, alle Transporte werden geprüft ...")
194 self.root.update_idletasks()
195
196 # Datenbankverbindung herstellen
197 conn = pyodbc.connect(verbindungs_i)
198 cursor = conn.cursor()
199
200 # Alle Transport-IDs aus der Datenbank holen
201 cursor.execute("SELECT DISTINCT transportid FROM dbo.coolchain ORDER BY transportid")
202 alle_ids = [row[0] for row in cursor.fetchall()]
203
204 cursor.close()
205 conn.close()
206
207 ergebnisse = []
208
209 # Für jede Transport-ID die Meldungen holen
210 for i, tid in enumerate(alle_ids, start=1):
211 transport_daten, _ = get_transport_daten(tid, verbindungs_i)
212 temperatur_daten, _ = get_temperatur_daten(transport_daten, verbindungs_i)
213 company_daten, _ = get_company_daten(transport_daten, verbindungs_i)
214 transportstation_daten, _ = get_transportstation_daten(transport_daten, verbindungs_i)
215
216 meldungen = verarbeite_transport(
217 transport_daten, temperatur_daten, company_daten, transportstation_daten
218 )
219
220 for m in meldungen:
221 ergebnisse.append((tid, m))
222
223
224 # Ergebnisse sortieren nach Meldungstext
225 ergebnisse.sort(key=lambda x: x[1])
226
227 # Ausgabe ins Textfeld
228 self.output.delete("1.0", tk.END)
229 self.output.insert(tk.END, "Alle Transport-IDs - Prüfungen\n")
230 self.output.insert(tk.END, "==================================================\n")
231
232 for tid, m in ergebnisse:
233 self.output.insert(tk.END, f"{tid} -> {m}\n")
234
235 self.status.set(f"{len(alle_ids)} Transporte geprüft")
236
237 except Exception as e:
238 messagebox.showerror("Fehler", f"Ein Fehler ist aufgetreten:\n{e}")
