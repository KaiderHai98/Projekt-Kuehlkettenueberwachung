@brief Führt die Prüfung aller in der Datenbank vorhandenen Transporte durch.

@details Ruft zunächst alle eindeutigen Transport-IDs ab. Iteriert dann
über jede ID, holt die zugehörigen Daten, verarbeitet sie und sammelt
alle generierten Meldungen. Die Ergebnisse werden nach Meldungstext sortiert
und im Ausgabefeld angezeigt. Stellt Status-Updates bereit.

175 '''
176 @brief Führt die Prüfung aller in der Datenbank vorhandenen Transporte durch.
177
178 @details Ruft zunächst alle eindeutigen Transport-IDs ab. Iteriert dann
179 über jede ID, holt die zugehörigen Daten, verarbeitet sie und sammelt

181 und im Ausgabefeld angezeigt. Stellt Status-Updates bereit.
182 '''
183
184 try:
185 # Status aktualisieren: Prüfung läuft

187 self.root.update_idletasks()

189 # Datenbankverbindung herstellen
190 conn = pyodbc.connect(verbindungs_i)
191 cursor = conn.cursor()
192
193 # Alle Transport-IDs aus der Datenbank holen
194 cursor.execute("SELECT DISTINCT transportid FROM dbo.coolchain ORDER BY transportid")
195 alle_ids = [row[0] for row in cursor.fetchall()]
196
197 cursor.close()
198 conn.close()
199
200 ergebnisse = []
201
202 # Für jede Transport-ID die Meldungen holen
203 for i, tid in enumerate(alle_ids, start=1):
204 transport_daten, _ = get_transport_daten(tid, verbindungs_i)
205 temperatur_daten, _ = get_temperatur_daten(transport_daten, verbindungs_i)
206 company_daten, _ = get_company_daten(transport_daten, verbindungs_i)
207 transportstation_daten, _ = get_transportstation_daten(transport_daten, verbindungs_i)
208
209 meldungen = verarbeite_transport(
210 transport_daten, temperatur_daten, company_daten, transportstation_daten
211 )
212
213 for m in meldungen:
214 ergebnisse.append((tid, m))
215
216
217 # Ergebnisse sortieren nach Meldungstext
218 ergebnisse.sort(key=lambda x: x[1])
219
220 # Ausgabe ins Textfeld
221 self.output.delete("1.0", tk.END)
222 self.output.insert(tk.END, "Alle Transport-IDs - Prüfungen\n")
223 self.output.insert(tk.END, "==================================================\n")
224
225 for tid, m in ergebnisse:
226 self.output.insert(tk.END, f"{tid} -> {m}\n")
227
228 self.status.set(f"{len(alle_ids)} Transporte geprüft")
